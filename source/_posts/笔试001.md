2.  请用javascript实现算法，寻找以下字符串中出现次数最多的、位置最靠前的字符

    ```js
    //给的字符串
    let strs = '4eqpjqpijwijsl';
    
    //解答
    let length = strs.length;
    
    let counter = {
        0: { index: length, count: 0 }
    }
    
    for (let i = 48; i < 58; i++) {
        counter[String.fromCharCode(i)] = { index: length, count: 0 }
    }
    
    for (let j = 97; j < 121; j++) {
        counter[String.fromCharCode(j)] = { index: length, count: 0 }
    }
    
    for (let i = 0; i < length; i++) {
        counter[strs[i]].count++;
        if (i < counter[strs[i]].index) {
            counter[strs[i]].index = i;
        }
    }
    
    let index = length, count = 0, charOne = "";
    
    for (let key in counter) {
        if (counter[key].count > count) {
            count = counter[key].count;
            index = counter[key].index;
            charOne = key;
        } else if (counter[key].count === count) {
            if (counter[key].index < index) {
                index = counter[key].index;
                charOne = key;
            }
        }
    }
    ```
